<body>

    <h1>關鍵詞篩選器</h1>

    <textarea id="inputText"></textarea><br>

    <button onclick="filterText()">篩選文本</button>

    <div class="container">

        <div class="box">

            <h2>篩選後的文本</h2>

            <pre id="filteredText"></pre>

        </div>

        <div class="box">

            <h2>篩選後詞頻統計</h2>

            <pre id="wordFrequency"></pre>

        </div>

    </div>

    

    <script>

        const stopwords = new Set(["是", "和", "與", "及", "在", "了", "的", "將", "於", "以", "並", "等", "後", "這", "那", "但", "也", "卻", "或", "若", "及其", "因為", "所以", "以及", "如果", "當", "則", "由於", "需要"]);

        

        function filterText() {

            let text = document.getElementById("inputText").value;

            let lines = text.split("\n");

            let processedLines = [];

            

            lines.forEach(line => {

                let cleanLine = line.replace(/[^\w\s.]/g, "").replace(/\s\d{1,4}(?=$|\s)/g, "");

                let words = cleanLine.split(" ").filter(word => word.length > 1 && !stopwords.has(word));

                processedLines.push(words.join(" "));

            });

            

            let filteredText = processedLines.join("\n");

            document.getElementById("filteredText").textContent = filteredText;

            

            let wordCounts = {};

            filteredText.split(/\s+/).forEach(word => {

                if (word) wordCounts[word] = (wordCounts[word] || 0) + 1;

            });

            

            let freqText = Object.entries(wordCounts).map(([word, count]) => `${word}: ${count}`).join("\n");

            document.getElementById("wordFrequency").textContent = freqText;

        }

    </script>
